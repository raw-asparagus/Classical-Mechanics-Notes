    \subsection{Elementary Scalar and Vector Operations}
    For vectors $\bm{A}$, $\bm{B}$ \& $\bm{C}$ (with components $A_i$, $B_i$ \& $C_i$ respectively) and scalars $\psi$, $\phi$ \& $\eta$\newline
    Addition
    \begin{dasheditemize}
        \item Vectors
        \begin{eqnindent}
            \begin{flalign}
                A_i + B_i &= B_i + A_i\quad\text{Commutative law} &&\\
                A_i + \paren{B_i + C_i} &= \paren{A_i + B_i} + C_i\quad\text{Associative law} &&
            \end{flalign}
        \end{eqnindent}
        \item Scalars
        \begin{eqnindent}
            \begin{flalign}
                \phi + \psi &= \psi + \phi\quad\text{Commutative law} &&\\
                \phi + \paren{\psi + \eta} &= \paren{\phi + \psi} + \eta\quad\text{Associative law} &&
            \end{flalign}
        \end{eqnindent}
    \end{dasheditemize}
    Multiplication by a scalar $\eta$
    \begin{dasheditemize}
        \item vector
        \begin{eqnindent}
            \begin{flalign}
                \eta\bm{A} = \bm{B} &&
            \end{flalign}
        \end{eqnindent}
        results in a vector
        \item scalar
        \begin{eqnindent}
            \begin{flalign}
                \eta\phi = \psi &&
            \end{flalign}
        \end{eqnindent}
        results in a scalar
    \end{dasheditemize}
